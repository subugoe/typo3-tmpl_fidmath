RTE.default.proc {
    // TRANSFORMATION METHOD
    overruleMode = ts_css

    // LINES CONVERSION
    dontConvBRtoParagraph = 1

    // TAGS ALLOWED OUTSIDE P & DIV
    allowTagsOutside = img,hr

    // TAGS ALLOWED
    allowTags = table, tbody, tr, th, td, h1, h2, h3, h4, h5, h6, div, p, br, span, ul, ol, li, re, blockquote, strong, em, b, i, u, sub, sup, strike, a, img, nobr, hr, tt, q, cite, abbr, acronym

    // TAGS DENIED
    denyTags = font

    // ALLOWED P & DIV ATTRIBUTES
    keepPDIVattribs = align,class,style

    // CONTENT TO RTE
    HTMLparser_rte {
        // TAGS ALLOWED
        allowTags < RTE.default.proc.allowTags
        removeTags = font

        // DO NOT REMOVE UNMATCHED TAGS
        keepNonMatchedTags = 0
    }

    // CONTENT TO DATABASE
    entryHTMLparser_db = 1
    entryHTMLparser_db {
        // XHTML COMPLIANCE
        xhtml_cleaning = 1

        // CLEAN TAGS

        noAttrib = b, i, u, strike, sub, sup, strong, em, quote, blockquote, cite, tt, br, center
        tags.hr.allowedAttribs = class
        // REMAP B AND I TAGS
        tags.b.remap = strong
        tags.i.remap = em

        rmTagIfNoAttrib = span,div,font

        htmlSpecialChars = 1
    }
}
